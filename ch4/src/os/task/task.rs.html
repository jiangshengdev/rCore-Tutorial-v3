1//! Types related to task management
2use super::TaskContext;
3use crate::config::{TRAP_CONTEXT, kernel_stack_position};
4use crate::mm::{KERNEL_SPACE, MapPermission, MemorySet, PhysPageNum, VirtAddr};
5use crate::trap::{TrapContext, trap_handler};
6
7/// task control block structure
8pub struct TaskControlBlock {
9    pub task_status: TaskStatus,
10    pub task_cx: TaskContext,
11    pub memory_set: MemorySet,
12    pub trap_cx_ppn: PhysPageNum,
13    #[allow(unused)]
14    pub base_size: usize,
15    pub heap_bottom: usize,
16    pub program_brk: usize,
17}
18
19impl TaskControlBlock {
20    pub fn get_trap_cx(&self) -> &'static mut TrapContext {
21        self.trap_cx_ppn.get_mut()
22    }
23    pub fn get_user_token(&self) -> usize {
24        self.memory_set.token()
25    }
26    pub fn new(elf_data: &[u8], app_id: usize) -> Self {
27        // memory_set with elf program headers/trampoline/trap context/user stack
28        let (memory_set, user_sp, entry_point) = MemorySet::from_elf(elf_data);
29        let trap_cx_ppn = memory_set
30            .translate(VirtAddr::from(TRAP_CONTEXT).into())
31            .unwrap()
32            .ppn();
33        let task_status = TaskStatus::Ready;
34        // map a kernel-stack in kernel space
35        let (kernel_stack_bottom, kernel_stack_top) = kernel_stack_position(app_id);
36        KERNEL_SPACE.exclusive_access().insert_framed_area(
37            kernel_stack_bottom.into(),
38            kernel_stack_top.into(),
39            MapPermission::R | MapPermission::W,
40        );
41        let task_control_block = Self {
42            task_status,
43            task_cx: TaskContext::goto_trap_return(kernel_stack_top),
44            memory_set,
45            trap_cx_ppn,
46            base_size: user_sp,
47            heap_bottom: user_sp,
48            program_brk: user_sp,
49        };
50        // prepare TrapContext in user space
51        let trap_cx = task_control_block.get_trap_cx();
52        *trap_cx = TrapContext::app_init_context(
53            entry_point,
54            user_sp,
55            KERNEL_SPACE.exclusive_access().token(),
56            kernel_stack_top,
57            trap_handler as usize,
58        );
59        task_control_block
60    }
61    /// change the location of the program break. return None if failed.
62    pub fn change_program_brk(&mut self, size: i32) -> Option<usize> {
63        let old_break = self.program_brk;
64        let new_brk = self.program_brk as isize + size as isize;
65        if new_brk < self.heap_bottom as isize {
66            return None;
67        }
68        let result = if size < 0 {
69            self.memory_set
70                .shrink_to(VirtAddr(self.heap_bottom), VirtAddr(new_brk as usize))
71        } else {
72            self.memory_set
73                .append_to(VirtAddr(self.heap_bottom), VirtAddr(new_brk as usize))
74        };
75        if result {
76            self.program_brk = new_brk as usize;
77            Some(old_break)
78        } else {
79            None
80        }
81    }
82}

85/// task status: UnInit, Ready, Running, Exited
86pub enum TaskStatus {
87    Ready,
88    Running,
89    Exited,
